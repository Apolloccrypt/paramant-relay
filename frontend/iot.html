// CI/CD INTEGRATIE

Secrets nooit in ENV of logs.

# GitHub Actions — secrets via Ghost Pipe
- name: Inject database wachtwoord
  uses: apolloccrypt/paramant-action@v1
  with:
    data:    ${{ secrets.DB_PASSWORD }}
    to:      ${{ secrets.SERVER_GP_PUBKEY }}
    api-key: ${{ secrets.PARAMANT_API_KEY }}

# Server ontvangt naar RAM-disk
paramant-cli receive \
  --hash "$HASH" \
  --out  /run/secrets/db_password
# → Nooit op schijf. Nooit in logs.
