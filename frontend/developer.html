Create a signing request
export PARAMANT_PARASIGN_KEY='psk_live_...'

curl -X POST https://paramant.app/v1/envelopes \
  -H "Authorization: Bearer $PARAMANT_PARASIGN_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "document": { "content_base64": "JVBERi0xLjc..." },
    "original_filename": "agreement-demo.pdf",
    "signers": [
      { "name": "Signer Demo", "email": "signer@example.com" }
    ]
  }'
